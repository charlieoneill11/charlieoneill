def plot_samples():
    fig, axs = plt.subplots(3,3)
    axs = axs.flatten()
    imgs = []
    for i in range(30, 39):
        imgs.append(np.squeeze(kmnist[i][0]))
    for img, ax in zip(imgs, axs):
        ax.imshow(img, cmap='gray', interpolation='none')
    plt.show()
